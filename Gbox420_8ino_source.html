203 static bool SyncInProgress = false;
204 time_t getNtpTime()
205 {
206  time_t NTPResponse = 0;
207  if (!SyncInProgress)
208  {
209  SyncInProgress = true;
210  uint32_t LastRefresh = millis();
211  logToSerials(F("Waiting for NTP time (15sec timeout)..."), false, 0);
212  while (NTPResponse == 0 && millis() - LastRefresh < 15000)
213  {
214  NTPResponse = ESPCmd.GetTime();
215  delay(500);
216  logToSerials(F("."), false, 0);
217  wdt_reset();
218  }
219  SyncInProgress = false;
220  if (NTPResponse == 0)
221  {
222  logToSerials(F("NTP time sync failed"), true, 1);
224  }
225  else
226  logToSerials(F("time synchronized"), true, 1);
227  }
228  return NTPResponse;
229 }
